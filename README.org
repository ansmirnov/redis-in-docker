** Select docker image version

#+BEGIN_SRC sh
IMAGE="redis"
TAG="6.0.10-buster"
#+END_SRC

** Pull docker image

#+BEGIN_SRC sh
docker pull $IMAGE:$TAG
#+END_SRC

** Run redis

#+BEGIN_SRC sh
docker run -d --net host --name redis $IMAGE:$TAG redis-server --bind 127.0.0.1
#+END_SRC

** Inspect the container

#+BEGIN_SRC sh
docker inspect redis
#+END_SRC

** Kill the container

#+BEGIN_SRC sh
docker kill redis
#+END_SRC
